cmake_minimum_required(VERSION 3.16)
set(PROJECT_VER "0.1.0")

set(PHOTOFRAME_FW_GIT_SHA "nogit")
execute_process(
  COMMAND git -C ${CMAKE_CURRENT_LIST_DIR} rev-parse --short=8 HEAD
  OUTPUT_VARIABLE PHOTOFRAME_FW_GIT_SHA_OUT
  RESULT_VARIABLE PHOTOFRAME_FW_GIT_SHA_RESULT
  OUTPUT_STRIP_TRAILING_WHITESPACE
)
if(PHOTOFRAME_FW_GIT_SHA_RESULT EQUAL 0 AND NOT PHOTOFRAME_FW_GIT_SHA_OUT STREQUAL "")
  set(PHOTOFRAME_FW_GIT_SHA "${PHOTOFRAME_FW_GIT_SHA_OUT}")
endif()

set(PHOTOFRAME_FW_VERSION "${PROJECT_VER}+${PHOTOFRAME_FW_GIT_SHA}")
execute_process(
  COMMAND git -C ${CMAKE_CURRENT_LIST_DIR} diff --quiet --exit-code
  RESULT_VARIABLE PHOTOFRAME_FW_GIT_DIRTY_RESULT
)
if(PHOTOFRAME_FW_GIT_DIRTY_RESULT EQUAL 1)
  set(PHOTOFRAME_FW_VERSION "${PHOTOFRAME_FW_VERSION}-dirty")
endif()

include($ENV{IDF_PATH}/tools/cmake/project.cmake)
project(photoframe_fw)
